export class Code extends LitElement {
  language = "js";

  constructor() {
    super();
    const highlightedCode = Prism.highlight(this.code ? this.code : '', Prism.languages.js, this.language);
  }

  private get code() {
    return this.shadowRoot?.querySelector("slot")?.firstChild?.textContent;
  }

  override render() {
    return html`
      <slot name="code"></slot>
    `
  }
}
